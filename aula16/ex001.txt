CREATE TABLE usuario (
pk_usuario SERIAL PRIMARY KEY,
nome VARCHAR(150) NOT NULL,
data_nascimento DATE NOT NULL,
endereco VARCHAR(200) NOT NULL,
telefone VARCHAR(15)
);

CREATE TABLE livro (
pk_livro SERIAL PRIMARY KEY,
titulo VARCHAR(150) NOT NULL,
genero VARCHAR(100) NOT NULL,
autor VARCHAR(150) NOT NULL,
data_lancamento DATE,
prateleira VARCHAR(15) NOT NULL,
total_copias INT NOT NULL
);

CREATE TABLE emprestimo (
pk_emprestimo SERIAL PRIMARY KEY,
fk_usuario INT NOT NULL,
fk_livro INT NOT NULL,
data_emprestimo DATE NOT NULL,
data_devolucao DATE NOT NULL,

CONSTRAINT emprestimo_usuario
FOREIGN KEY (fk_usuario)
REFERENCES usuario(pk_usuario)
ON DELETE CASCADE,

CONSTRAINT emprestimo_livro
FOREIGN KEY (fk_livro)
REFERENCES livro(pk_livro)
ON DELETE CASCADE
);
